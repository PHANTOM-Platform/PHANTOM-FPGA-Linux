bitfileexists=test -e mmc 0 /fpga/bitfile/bitstream.bit
fpgaload=if run bitfileexists; then load mmc 0 0x1000000 /fpga/bitfile/bitstream.bit; fpga loadb 0 0x1000000 ${filesize}; fi
bootargs=console=ttyPS0,115200 root=/dev/mmcblk0p2 rw rootwait devtmpfs.mount=0 uio_pdrv_genirq.of_id=phantom_platform,generic-uio,ui_pdrv
uenvcmd=run fpgaload; load mmc 0 ${kernel_load_address} ${kernel_image} && load mmc 0 ${devicetree_load_address} ${devicetree_image} && bootm ${kernel_load_address} - ${devicetree_load_address}
